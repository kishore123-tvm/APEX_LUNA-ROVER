            #include <LiquidCrystal.h>
            #include <SoftwareSerial.h>
            LiquidCrystal lcd(13, 12, 11, 10, 9, 8);
            SoftwareSerial ble(2, 3);
            #define splash splash1
            
            String voice;
            const int m1 =  4;
            const int m2 = 5;
            const int m3 = 6;
            const int m4 = 7;
            
            float v1, v2, v3;
            
            void setup() {
              // put your setup code here, to run once:

              pinMode(m2, OUTPUT);

              pinMode(m4, OUTPUT);

            void LcDSet() {
              lcd.begin(16, 2);
            
              splash(0, "Combat Robot");
              splash(1, "");
              delay(2000);
              lcd.clear();
            }
            
            
            void loop() {
              // put your main code here, to run repeatedly:

            void robo()
            {
            
              while (ble.available()) { //Check if there is an available byte to read
                lcd.clear();
                delay(10); //Delay added to make thing stable
                char c = ble.read(); //Conduct a serial read
                //    Serial.println(c);
                if (c == '#')
                {
                  break; //Exit the loop when the # is detected after the word
                }
                voice += c; //Shorthand for voice = voice + c
              }
              if (voice.length() > 0) {
                //Serial.println(voice);
            
                if (voice == "*forward") {
                  lcd.setCursor(0, 0);
                  lcd.print("-----FORWARD------");
                  forward();
            
                  //      left();
                }
                else if (voice == "*backward") {
                  lcd.setCursor(0, 0);
                  lcd.print("---BACKWARD------");
                  backward();
                  //      right();
                }
            
                //----------Turn On One-By-One----------//
                else if (voice == "*right") {
                  lcd.setCursor(0, 0);
                  lcd.print("-----RIGHT------");
                  right();
                  //      forward();
                  // left();
                }
                else if (voice == "*left") {
                  lcd.setCursor(0, 0);
                  lcd.print("-----LEFT------");
                  //right();
                  left();
                  //      backward();
                }
            
            
            
            
            
                else if (voice == "*stop") {
                  lcd.setCursor(0, 0);
                  lcd.print("-----Stop------");
                  stopbot();
                }
            
            
                voice = "";
            
              }
              //  delay(300);
            
            }
            void forward()
            {
              Serial.println("FORWARD" );
              digitalWrite(m1, HIGH); digitalWrite(m2, LOW);
              digitalWrite(m3, HIGH); digitalWrite(m4, LOW);
            }
            
            void backward()
            {
              Serial.println("BACKWARD" );
              digitalWrite(m1, LOW); digitalWrite(m2, HIGH);
              digitalWrite(m3, LOW); digitalWrite(m4, HIGH);
            }
            
            void right()
            {
              Serial.println("RIGHT" );
              digitalWrite(m1, HIGH); digitalWrite(m2, LOW);
              digitalWrite(m3, LOW); digitalWrite(m4, HIGH);
            
              delay(1500);
              stopbot();
            }
            
            void left()
            {
              Serial.println("LEFT" );
              digitalWrite(m1, LOW); digitalWrite(m2, HIGH);
              digitalWrite(m3, HIGH); digitalWrite(m4, LOW);
            
              delay(1500);
              stopbot();
            }
            
            
            void stopbot()
            {
              Serial.println("STOP" );
              digitalWrite(m1, LOW); digitalWrite(m2, LOW);
              digitalWrite(m3, LOW); digitalWrite(m4, LOW);
            }
